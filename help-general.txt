SELECT cl.clientid as Client , 
	cl.fullname as FullName , 
	pr.productid as Product , 
	pr.description as Description , 
	pr.price as Price , 
	clpr.clientproductid as Receipt , 
	clpr.amount as Debt , 
	clpr.status as Status 
FROM schmicroservices.tbl_client cl 
INNER JOIN schmicroservices.tbl_client_product clpr ON clpr.clientid = cl.clientid 
INNER JOIN schmicroservices.tbl_product pr ON pr.productid = clpr.productid 
AND cl.clientid = 1 
AND cl.companyid = 1
